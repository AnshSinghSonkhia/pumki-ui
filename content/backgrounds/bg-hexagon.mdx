# Hexagon Background 

An unique interactive hexagon grid background component for modern applications, with customizable colors and patterns.

<Preview 
  title="Tiny Honeycomb"
  description="Dense pattern with small hexagons creating a subtle texture like honeycomb."
  code={`'use client';
 
import { HexagonBackground } from '@/components/pumki-ui/backgrounds/bg-hexagon';

export const TinyHoneycombDemo = () => {
    return (
        <HexagonBackground
            hexagonSize={25}
            hexagonMargin={1}
            className="absolute inset-0 flex items-center justify-center rounded-xl bg-gradient-to-br from-amber-300 via-yellow-400 to-amber-500"
        />
    );
};`}
>
    <HexagonBackground
        hexagonSize={25}
        hexagonMargin={1}
        className="absolute inset-0 flex items-center justify-center rounded-xl bg-gradient-to-br from-amber-300 via-yellow-400 to-amber-500"
    />
</Preview>

<Callout>The background is interactive, allowing users to engage with the visual elements.</Callout>
