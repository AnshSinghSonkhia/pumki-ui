# Draw Line Text Animation

A smooth text effect that draws characters as strokes and fills them in, perfect for headers and highlights.

<Callout>Perfect for headers and highlights.</Callout>

<Preview
  code={`import { DrawLineText } from '@/components/pumki-ui/text/draw-line-text';

  export const DrawLineTextDemo = () => {
    return (
        <DrawLineText
            className="font-medium"
            fontSize={60}
            strokeWidth={1.5}
            text="Ansh Singh Sonkhia"
            color="var(--color-foreground)"
        />
    );
};`}
>
        <DrawLineText
            className="font-medium"

            strokeWidth={1.5}

            color="var(--color-foreground)"
        />
