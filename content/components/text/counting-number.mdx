# Counting Number Animation

A number display component that animates values with a smooth counting effect, perfect for stats, metrics, and dashboards.

<Preview
  code={`import { CountingNumber } from '@/components/pumki-ui/text/counting-number';

  export const CountingNumberDemo = () => {
    return (
        <CountingNumber number={new Date().getFullYear()} className="text-4xl" />
    );
};`}
>
        <CountingNumber number={new Date().getFullYear()} className="text-4xl" />

</Preview>

## Example

<Preview
  title="Using Decimals"
  code={`import { CountingNumber } from '@/components/pumki-ui/text/counting-number';

  export const CountingNumberDemo = () => {
    return (
        <CountingNumber
            number={12345.67}
            decimalPlaces={2}
            decimalSeparator=","
            className="text-4xl"
            inView
        />
    );
};`}
>
        <CountingNumber
            number={12345.67}
            decimalPlaces={2}
            decimalSeparator=","
            className="text-4xl"
            inView
        />

</Preview>

<Preview
  title="Starting from a specific number"
  description="You can start counting from a specific number by using the `fromNumber` prop."
  code={`import { CountingNumber } from '@/components/pumki-ui/text/counting-number';

  export const CountingNumberDemo = () => {
    return (
        <CountingNumber
            number={new Date().getFullYear()}
            fromNumber={0}
            className="text-4xl"
            inView
        />
    );
};`}
>
        <CountingNumber
            number={new Date().getFullYear()}
            fromNumber={0}
            className="text-4xl"
            inView
        />

</Preview>
