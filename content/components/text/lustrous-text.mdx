# Lustrous Text Animation

A component that animates the lustrous effect on text.

## Example

<Preview
  title="Gold Customizations"
  code={`import { LustrousText } from '@/components/pumki-ui/text/lustrous-text';

  export const LustrousTextDemo = () => {
    return (
        <LustrousText
            className="text-4xl font-semibold tracking-wide"
            text="Ansh Singh Sonkhia India"
            wave
            lustrousColor="#FFD700"
            color="#B8860B"
            duration={1.5}
        />
    );
};`}
>
    <LustrousText
        className="text-4xl font-semibold tracking-wide"
        text="Ansh Singh Sonkhia India"
        wave
        lustrousColor="#FFD700"
        color="#B8860B"
        duration={1.5}
    />

</Preview>
