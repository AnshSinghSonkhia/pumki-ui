# Sliding Number Animation

A component that animates the transition between numbers.

<Callout>Also used as Counter</Callout>

<Preview
  code={`import { SlidingNumber } from '@/components/pumki-ui/text/sliding-number'

  export const SlidingNumberDemo = () => {
    return (
        <SlidingNumber
            number={12345.67}
            decimalSeparator="."
            decimalPlaces={2}
            padStart
            className="text-4xl"
            inView
        />
    );
};`}
>
    <SlidingNumber
      number={12345.67}
      decimalSeparator="."
      decimalPlaces={2}
      padStart
      className="text-4xl"
      inView
    />

</Preview>
